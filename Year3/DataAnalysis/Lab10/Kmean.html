In[]:
# To find the best value for k, we will use the elbow method
k_values = range(1, 10)
inertia_values = []
for k in k_values:
    model = KMeans(n_clusters=k, random_state=42)
    model.fit(X_norm)
    inertia_values.append(model.inertia_)

sns.lineplot(x=k_values, y=inertia_values).set_title("Elbow Method")
